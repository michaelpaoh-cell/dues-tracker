/** * Google Apps Script - Backend for Monthly Dues Tracker v2 * Adds: Member details (position, duty, address), Due date + reminder support */ function setupSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); if (!ss.getSheetByName('Members')) { const sh = ss.insertSheet('Members'); sh.appendRow(['id', 'name', 'position', 'duty', 'address']); } else { const sh = ss.getSheetByName('Members'); const headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; if (headers.indexOf('position') === -1) { sh.getRange(1, headers.length+1, 1, 3).setValues([['position','duty','address']]); } } if (!ss.getSheetByName('Dues')) { const sh = ss.insertSheet('Dues'); sh.appendRow(['month', 'memberId', 'amount', 'paid', 'dueDate']); } else { const sh = ss.getSheetByName('Dues'); const headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; if (headers.indexOf('dueDate') === -1) { sh.getRange(1, headers.length+1, 1, 1).setValues([['dueDate']]); } } if (!ss.getSheetByName('Expenses')) { const sh = ss.insertSheet('Expenses'); sh.appendRow(['id', 'title', 'amount', 'date']); } } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action; if (action === 'getAll') { return jsonResponse(getAllData(ss)); } return jsonResponse({ error: 'unknown action' }); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const body = JSON.parse(e.postData.contents); const action = body.action; if (action === 'syncAll') { fullSync(ss, body.data); return jsonResponse({ success: true }); } return jsonResponse({ error: 'unknown action' }); } function getAllData(ss) { const members = sheetToObjects(ss.getSheetByName('Members')); const duesRaw = sheetToObjects(ss.getSheetByName('Dues')); const expenses = sheetToObjects(ss.getSheetByName('Expenses')); const dues = {}; duesRaw.forEach(r => { if (!dues[r.month]) dues[r.month] = { amount: Number(r.amount) || 0, dueDate: r.dueDate || '', paid: {} }; dues[r.month].paid[r.memberId] = (r.paid === true || r.paid === 'TRUE' || r.paid === 'true'); if (r.amount) dues[r.month].amount = Number(r.amount); if (r.dueDate) dues[r.month].dueDate = r.dueDate; }); return { members, dues, expenses }; } function sheetToObjects(sheet) { const rows = sheet.getDataRange().getValues(); const headers = rows.shift(); return rows.filter(r => r[0] !== '').map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); } function fullSync(ss, data) { const memSheet = ss.getSheetByName('Members'); memSheet.clearContents(); memSheet.appendRow(['id', 'name', 'position', 'duty', 'address']); data.members.forEach(m => memSheet.appendRow([m.id, m.name, m.position||'', m.duty||'', m.address||''])); const duesSheet = ss.getSheetByName('Dues'); duesSheet.clearContents(); duesSheet.appendRow(['month', 'memberId', 'amount', 'paid', 'dueDate']); for (const month in data.dues) { const rec = data.dues[month]; for (const memberId in rec.paid) { duesSheet.appendRow([month, memberId, rec.amount, rec.paid[memberId], rec.dueDate || '']); } } const expSheet = ss.getSheetByName('Expenses'); expSheet.clearContents(); expSheet.appendRow(['id', 'title', 'amount', 'date']); data.expenses.forEach(e => expSheet.appendRow([e.id, e.title, e.amount, e.date])); } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
